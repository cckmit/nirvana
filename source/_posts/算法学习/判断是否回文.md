/**
 * 判断是否回文
**/
function checkPlalindrome(str){
  if(!str){//为空，或者长度能整除2
    return false
  }
  
  let isOdd = str.length % 2;
  let mid = Math.floor(str.length/2);//中间数
  let flag = true;
  for (let i = 0;i<mid;i++){
    console.log(str.substr(i,1),str.substr(i+mid+isOdd,1))
    if(!(str.substr(i,1) == str.substr(i+mid+isOdd,1))){
      flag = false;
      break;
    }
  }
  return flag;
}
